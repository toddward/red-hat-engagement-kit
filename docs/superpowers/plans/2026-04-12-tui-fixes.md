### Task 1: Fix Menu Back-Navigation (Esc from Submenus)

**Problem:** When you select "Agents" or "Skills" from the main menu, pressing Esc does nothing because `currentView` is still `ViewMenu`. The menu items change but there's no way to go back.

**Fix:** Add a menu history stack. When entering a submenu, push the current menu. On Esc, pop back.

**Files:**
- Modify: `tui/viewer/ui/menu.go`
- Modify: `tui/viewer/ui/app.go`

- [ ] **Step 1: Add menu stack to Menu**

In `tui/viewer/ui/menu.go`, add a `history` field and push/pop methods:

After the `Menu` struct definition, add:

```go
// PushMenu saves current menu state and replaces with new items
func (m *Menu) PushMenu(title string, items []MenuItem) {
	prev := MenuState{
		title:  m.title,
		items:  m.items,
		cursor: m.cursor,
	}
	m.history = append(m.history, prev)
	m.title = title
	m.items = items
	m.cursor = 0
}

// PopMenu restores the previous menu. Returns false if at root.
func (m *Menu) PopMenu() bool {
	if len(m.history) == 0 {
		return false
	}
	prev := m.history[len(m.history)-1]
	m.history = m.history[:len(m.history)-1]
	m.title = prev.title
	m.items = prev.items
	m.cursor = prev.cursor
	return true
}

// AtRoot returns true if there's no parent menu to return to.
func (m Menu) AtRoot() bool {
	return len(m.history) == 0
}
```

Add the `MenuState` type and `history` field to the `Menu` struct:

```go
type MenuState struct {
	title  string
	items  []MenuItem
	cursor int
}

type Menu struct {
	items   []MenuItem
	cursor  int
	title   string
	width   int
	height  int
	history []MenuState
}
```

Remove the unused `parent *Menu` field.

```go
package ui

import (
	"strings"

	"github.com/charmbracelet/bubbles/viewport"
	tea "github.com/charmbracelet/bubbletea"
	"github.com/charmbracelet/lipgloss"
	"github.com/toddward/red-hat-engagement-kit/tui/viewer/protocol"
)

// ArtifactBrowser displays engagement artifacts
type ArtifactBrowser struct {
	tree     []protocol.ArtifactNode
	flat     []flatNode // flattened tree for navigation
	cursor   int
	viewport viewport.Model
	content  string // file content being viewed
	viewing  bool   // true when viewing file content
	width    int
	height   int
}

type flatNode struct {
	node  protocol.ArtifactNode
	depth int
}

func NewArtifactBrowser() ArtifactBrowser {
	return ArtifactBrowser{
		viewport: viewport.New(80, 20),
	}
}

func (ab *ArtifactBrowser) SetSize(width, height int) {
	ab.width = width
	ab.height = height
	ab.viewport.Width = width
	ab.viewport.Height = height - 6
}

func (ab *ArtifactBrowser) SetTree(tree []protocol.ArtifactNode) {
	ab.tree = tree
	ab.flat = flattenTree(tree, 0)
	ab.cursor = 0
	ab.viewing = false
}

func (ab *ArtifactBrowser) SetContent(content string) {
	ab.content = content
	ab.viewing = true
	ab.viewport.SetContent(content)
	ab.viewport.GotoTop()
}

func (ab *ArtifactBrowser) CloseContent() {
	ab.viewing = false
	ab.content = ""
}

func (ab ArtifactBrowser) SelectedPath() string {
	if ab.cursor < 0 || ab.cursor >= len(ab.flat) {
		return ""
	}
	return ab.flat[ab.cursor].node.Path
}

func (ab ArtifactBrowser) SelectedIsFile() bool {
	if ab.cursor < 0 || ab.cursor >= len(ab.flat) {
		return false
	}
	return ab.flat[ab.cursor].node.Type == "file"
}

func (ab ArtifactBrowser) IsViewing() bool {
	return ab.viewing
}

func flattenTree(nodes []protocol.ArtifactNode, depth int) []flatNode {
	var result []flatNode
	for _, n := range nodes {
		result = append(result, flatNode{node: n, depth: depth})
		if len(n.Children) > 0 {
			result = append(result, flattenTree(n.Children, depth+1)...)
		}
	}
	return result
}

func (ab ArtifactBrowser) Update(msg tea.Msg) (ArtifactBrowser, tea.Cmd) {
	if ab.viewing {
		var cmd tea.Cmd
		ab.viewport, cmd = ab.viewport.Update(msg)
		return ab, cmd
	}

	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.String() {
		case "up", "k":
			if ab.cursor > 0 {
				ab.cursor--
			}
		case "down", "j":
			if ab.cursor < len(ab.flat)-1 {
				ab.cursor++
			}
		}
	}
	return ab, nil
}

func (ab ArtifactBrowser) View() string {
	var b strings.Builder

	b.WriteString(TitleStyle.Render("Artifacts"))
	b.WriteString("\n")
	maxWidth := ab.width - 4
	if maxWidth > 40 {
		maxWidth = 40
	}
	if maxWidth < 1 {
		maxWidth = 1
	}
	b.WriteString(lipgloss.NewStyle().
		Foreground(RedHatRed).
		Render(strings.Repeat("━", maxWidth)))
	b.WriteString("\n\n")

	if ab.viewing {
		b.WriteString(ab.viewport.View())
		b.WriteString("\n\n")
		b.WriteString(HelpKeyStyle.Render("↑/↓"))
		b.WriteString(HelpDescStyle.Render(" scroll  "))
		b.WriteString(HelpKeyStyle.Render("Esc"))
		b.WriteString(HelpDescStyle.Render(" back"))
	} else {
		if len(ab.flat) == 0 {
			b.WriteString(lipgloss.NewStyle().Foreground(TextMuted).Italic(true).
				Render("No artifacts found"))
		} else {
			for i, fn := range ab.flat {
				indent := strings.Repeat("  ", fn.depth)
				cursor := "  "
				if i == ab.cursor {
					cursor = "> "
				}

				style := ArtifactFileStyle
				prefix := "  "
				if fn.node.Type == "directory" {
					style = ArtifactDirStyle
					prefix = "📁 "
				}

				if i == ab.cursor {
					style = style.Foreground(RedHatRed).Bold(true)
				}

				b.WriteString(cursor + indent + style.Render(prefix+fn.node.Name))
				b.WriteString("\n")
			}
		}
		b.WriteString("\n")
		b.WriteString(HelpKeyStyle.Render("↑/↓"))
		b.WriteString(HelpDescStyle.Render(" navigate  "))
		b.WriteString(HelpKeyStyle.Render("enter"))
		b.WriteString(HelpDescStyle.Render(" open  "))
		b.WriteString(HelpKeyStyle.Render("Esc"))
		b.WriteString(HelpDescStyle.Render(" back"))
	}

	return MainStyle.Render(b.String())
}
```

Create file `tui/viewer/ui/checklists.go`:

```go
package ui

import (
	"fmt"
	"strings"

	tea "github.com/charmbracelet/bubbletea"
	"github.com/charmbracelet/lipgloss"
	"github.com/toddward/red-hat-engagement-kit/tui/viewer/protocol"
)

// ChecklistBrowser displays and manages checklists
type ChecklistBrowser struct {
	// List mode
	names    []string
	listCursor int

	// Detail mode
	checklist *protocol.Checklist
	items     []checklistFlatItem
	cursor    int
	viewing   bool

	width  int
	height int
}

type checklistFlatItem struct {
	isHeader bool
	title    string
	item     *protocol.ChecklistItem
}

func NewChecklistBrowser() ChecklistBrowser {
	return ChecklistBrowser{}
}

func (cb *ChecklistBrowser) SetSize(width, height int) {
	cb.width = width
	cb.height = height
}

func (cb *ChecklistBrowser) SetNames(names []string) {
	cb.names = names
	cb.listCursor = 0
	cb.viewing = false
}

func (cb *ChecklistBrowser) SetChecklist(cl protocol.Checklist) {
	cb.checklist = &cl
	cb.viewing = true
	cb.cursor = 0

	cb.items = nil
	for _, section := range cl.Sections {
		cb.items = append(cb.items, checklistFlatItem{isHeader: true, title: section.Title})
		for i := range section.Items {
			cb.items = append(cb.items, checklistFlatItem{item: &section.Items[i]})
		}
	}
}

func (cb ChecklistBrowser) IsViewing() bool {
	return cb.viewing
}

func (cb ChecklistBrowser) SelectedName() string {
	if cb.listCursor < 0 || cb.listCursor >= len(cb.names) {
		return ""
	}
	return cb.names[cb.listCursor]
}

func (cb ChecklistBrowser) SelectedItem() *protocol.ChecklistItem {
	if cb.cursor < 0 || cb.cursor >= len(cb.items) {
		return nil
	}
	return cb.items[cb.cursor].item
}

func (cb *ChecklistBrowser) CloseDetail() {
	cb.viewing = false
	cb.checklist = nil
}

func (cb ChecklistBrowser) Update(msg tea.Msg) (ChecklistBrowser, tea.Cmd) {
	switch msg := msg.(type) {
	case tea.KeyMsg:
		switch msg.String() {
		case "up", "k":
			if cb.viewing {
				if cb.cursor > 0 {
					cb.cursor--
					// Skip headers
					if cb.items[cb.cursor].isHeader && cb.cursor > 0 {
						cb.cursor--
					}
				}
			} else {
				if cb.listCursor > 0 {
					cb.listCursor--
				}
			}
		case "down", "j":
			if cb.viewing {
				if cb.cursor < len(cb.items)-1 {
					cb.cursor++
					// Skip headers
					if cb.items[cb.cursor].isHeader && cb.cursor < len(cb.items)-1 {
						cb.cursor++
					}
				}
			} else {
				if cb.listCursor < len(cb.names)-1 {
					cb.listCursor++
				}
			}
		}
	}
	return cb, nil
}

func (cb ChecklistBrowser) View() string {
	var b strings.Builder

	title := "Checklists"
	if cb.viewing && cb.checklist != nil {
		title = cb.checklist.Name
	}
	b.WriteString(TitleStyle.Render(title))
	b.WriteString("\n")
	maxWidth := cb.width - 4
	if maxWidth > 40 {
		maxWidth = 40
	}
	if maxWidth < 1 {
		maxWidth = 1
	}
	b.WriteString(lipgloss.NewStyle().
		Foreground(RedHatRed).
		Render(strings.Repeat("━", maxWidth)))
	b.WriteString("\n\n")

	if cb.viewing && cb.checklist != nil {
		// Progress bar
		pct := cb.checklist.CompletionPct
		b.WriteString(ChecklistProgressStyle.Render(fmt.Sprintf("Progress: %d%%", pct)))
		b.WriteString("\n\n")

		for i, fi := range cb.items {
			if fi.isHeader {
				b.WriteString(MenuHeaderStyle.Render(fi.title))
				b.WriteString("\n")
				continue
			}

			cursor := "  "
			if i == cb.cursor {
				cursor = "> "
			}

			checkbox := "[ ] "
			style := ChecklistUncheckedStyle
			if fi.item.Checked {
				checkbox = "[x] "
				style = ChecklistCheckedStyle
			}

			if i == cb.cursor {
				style = style.Bold(true)
			}

			b.WriteString(cursor + style.Render(checkbox+fi.item.Text))
			b.WriteString("\n")
		}

		b.WriteString("\n")
		b.WriteString(HelpKeyStyle.Render("↑/↓"))
		b.WriteString(HelpDescStyle.Render(" navigate  "))
		b.WriteString(HelpKeyStyle.Render("space"))
		b.WriteString(HelpDescStyle.Render(" toggle  "))
		b.WriteString(HelpKeyStyle.Render("Esc"))
		b.WriteString(HelpDescStyle.Render(" back"))
	} else {
		if len(cb.names) == 0 {
			b.WriteString(lipgloss.NewStyle().Foreground(TextMuted).Italic(true).
				Render("No checklists found"))
		} else {
			for i, name := range cb.names {
				cursor := "  "
				style := MenuItemStyle
				if i == cb.listCursor {
					cursor = "> "
					style = MenuItemSelectedStyle
				}
				b.WriteString(style.Render(cursor + name))
				b.WriteString("\n")
			}
		}

		b.WriteString("\n")
		b.WriteString(HelpKeyStyle.Render("↑/↓"))
		b.WriteString(HelpDescStyle.Render(" navigate  "))
		b.WriteString(HelpKeyStyle.Render("enter"))
		b.WriteString(HelpDescStyle.Render(" open  "))
		b.WriteString(HelpKeyStyle.Render("Esc"))
		b.WriteString(HelpDescStyle.Render(" back"))
	}

	return MainStyle.Render(b.String())
}
```

```go
package ui

import (
	"strings"

	"github.com/charmbracelet/lipgloss"

// HelpView renders the keyboard shortcut help overlay
func HelpView() string {
	var b strings.Builder

	b.WriteString(TitleStyle.Render("Keyboard Shortcuts"))
	b.WriteString("\n\n")

	shortcuts := []struct {
		key  string
		desc string
	}{
		{"↑/↓ or j/k", "Navigate items"},
		{"Enter", "Select / confirm"},
		{"Esc", "Go back / close"},
		{"1-5", "Jump to menu item"},
		{"/", "Open command palette"},
		{"?", "Toggle this help"},
		{"Space", "Toggle checklist item"},
		{"q", "Quit (from main menu)"},
		{"Ctrl+C", "Cancel running execution"},
	}

	for _, s := range shortcuts {
		key := HelpKeyStyle.Width(16).Render(s.key)
		desc := HelpDescStyle.Render(s.desc)
		b.WriteString("  " + key + desc + "\n")
	}

	b.WriteString("\n")
	b.WriteString(lipgloss.NewStyle().Foreground(TextDim).Italic(true).
		Render("Press ? or Esc to close"))

	return PaletteStyle.Width(50).Render(b.String())
}
```

```go
func (a *App) handleBashResponse(resp protocol.Response) tea.Cmd {
    cmdName := a.pendingCmds[resp.ID]
    delete(a.pendingCmds, resp.ID)

    switch resp.Type {
    case "response":
        switch cmdName {
        case "init":
            var initResp protocol.InitResponse
            if err := json.Unmarshal(resp.Payload, &initResp); err == nil {
                a.skills = initResp.Skills
                a.engagements = initResp.Engagements
                a.agents = initResp.Agents
                if len(a.engagements) == 1 {
                    a.engagement = a.engagements[0].Slug
                    a.sidebar.SetEngagement(a.engagement)
                    return a.sendCommand("get_phase", map[string]string{"engagement": a.engagement})
                }
                a.buildPaletteItems()
            }
        case "get_phase":
            var phaseInfo protocol.PhaseInfo
            if err := json.Unmarshal(resp.Payload, &phaseInfo); err == nil {
                a.sidebar.SetPhase(phaseInfo)
            }
        case "list_artifacts":
            var treeResp struct {
                Tree []protocol.ArtifactNode `json:"tree"`
            }
            if err := json.Unmarshal(resp.Payload, &treeResp); err == nil {
                a.artifacts.SetTree(treeResp.Tree)
            }
        case "read_artifact":
            var contentResp struct {
                Content string `json:"content"`
            }
            if err := json.Unmarshal(resp.Payload, &contentResp); err == nil {
                a.artifacts.SetContent(contentResp.Content)
            }
        case "list_checklists":
            var clResp struct {
                Checklists []struct {
                    Name string `json:"name"`
                } `json:"checklists"`
            }
            if err := json.Unmarshal(resp.Payload, &clResp); err == nil {
                names := make([]string, len(clResp.Checklists))
                for i, cl := range clResp.Checklists {
                    names[i] = cl.Name
                }
                a.checklists.SetNames(names)
            }
        case "get_checklist":
            var cl protocol.Checklist
            if err := json.Unmarshal(resp.Payload, &cl); err == nil && cl.Name != "" {
                a.checklists.SetChecklist(cl)
            }
        case "set_state":
            // No-op, success
        }

    case "event":
        var event protocol.StreamEvent
        if err := json.Unmarshal(resp.Payload, &event); err == nil {
            if event.Event == protocol.EventQuestion {
                a.input.SetPrompt(event.Text, event.Options)
                a.currentView = ViewInput
            } else {
                a.activity.AddEvent(event)
            }
        }

    case "error":
        var errPayload struct {
            Message string `json:"message"`
        }
        if err := json.Unmarshal(resp.Payload, &errPayload); err == nil {
            a.activity.AddEvent(protocol.StreamEvent{
                Event: protocol.EventError,
                Text:  errPayload.Message,
            })
        }
    }

    return nil
}
```
